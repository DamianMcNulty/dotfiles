#centos rpm from source for ruby and other software

sudo yum install rpm-build redhat-rpm-config rpmdevtools mock
sudo yum --enablerepo epel install mock

sudo adduser builder --home-dir /home/builder --create-home --user-group --groups mock --shell /bin/bash --comment "rpm package builder"
su -l builder

rpmdev-setuptree

mkdir -p ~/rpmbuild/{BUILD,RPMS,SOURCES,SPECS,SRPMS}
echo '%_topdir %(echo $HOME)/rpmbuild' > ~/.rpmmacros

cd ~/rpmbuild/SOURCES
wget http://www.mirrorservice.org/sites/ftp.ruby-lang.org/pub/ruby/ruby-1.9.3-p429.tar.gz

cd ~
mock --init
cd ~/rpmbuild/SPECS
wget https://raw.github.com/imeyer/ruby-1.9.3-rpm/master/ruby19.spec
# Build the source package
mock --buildsrpm --spec=./rpmbuild/SPECS/ruby193.spec --sources=./rpmbuild/SOURCES
# Build the binary packages
mock --no-clean --rebuild /var/lib/mock/epel-6-i386/result/ruby-1.9.3p429-1.el6.src.rpm
